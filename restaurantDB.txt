CREATE DATABASE IF NOT EXISTS restaurantDB; -- Change to your database name
USE restaurantDB;

CREATE TABLE IF NOT EXISTS Menu (
  item_id VARCHAR(6) PRIMARY KEY,
  item_name VARCHAR(255),
  item_type VARCHAR(255),
  item_category VARCHAR(255),
  item_price DECIMAL(10, 2),
  item_description VARCHAR(255)
);

CREATE TABLE IF NOT EXISTS Accounts (
  account_id INT PRIMARY KEY AUTO_INCREMENT,
  email VARCHAR(255) NOT NULL,
  register_date DATE NOT NULL,
  phone_number VARCHAR(255) NOT NULL,
  password VARCHAR(255) NOT NULL,
  UNIQUE KEY uniq_accounts_email (email),
  UNIQUE KEY uniq_accounts_phone (phone_number)
);

CREATE TABLE IF NOT EXISTS Staffs (
  staff_id INT PRIMARY KEY AUTO_INCREMENT,
  staff_name VARCHAR(255) NOT NULL,
  role VARCHAR(255) NOT NULL,
  account_id INT NOT NULL,
  UNIQUE KEY uniq_staff_account (account_id),
  FOREIGN KEY (account_id ) REFERENCES Accounts(account_id )
);

CREATE TABLE IF NOT EXISTS Memberships (
  member_id INT PRIMARY KEY AUTO_INCREMENT,
  member_name VARCHAR(255) NOT NULL,
  points INT NOT NULL DEFAULT 0,
  account_id INT NOT NULL,
  UNIQUE KEY uniq_membership_account (account_id),
  CONSTRAINT chk_membership_points_nonnegative CHECK (points >= 0),
  FOREIGN KEY (account_id ) REFERENCES Accounts(account_id )
);

CREATE TABLE IF NOT EXISTS Restaurant_Tables (
  table_id INT PRIMARY KEY AUTO_INCREMENT,
  capacity INT NOT NULL,
  is_available BOOLEAN NOT NULL DEFAULT TRUE,
  CONSTRAINT chk_table_capacity_positive CHECK (capacity > 0)
);


CREATE TABLE IF NOT EXISTS Table_Availability (
  availability_id INT PRIMARY KEY AUTO_INCREMENT,
  table_id INT NOT NULL,
  reservation_date DATE NOT NULL,
  reservation_time TIME NOT NULL,
  status VARCHAR(20) NOT NULL DEFAULT 'reserved',
  UNIQUE KEY uniq_table_availability_slot (table_id, reservation_date, reservation_time),
  CONSTRAINT chk_table_availability_status CHECK (status IN ('reserved', 'available')),
  FOREIGN KEY (table_id) REFERENCES Restaurant_Tables(table_id)
);

CREATE TABLE IF NOT EXISTS Reservations (
  reservation_id INT PRIMARY KEY AUTO_INCREMENT,
  customer_name VARCHAR(255) NOT NULL,
  table_id INT NOT NULL,
  reservation_time TIME NOT NULL,
  reservation_date DATE NOT NULL,
  head_count INT NOT NULL,
  special_request VARCHAR(255),
  UNIQUE KEY uniq_reservation_slot (table_id, reservation_date, reservation_time),
  CONSTRAINT chk_reservation_head_count_positive CHECK (head_count > 0),
  FOREIGN KEY (table_id) REFERENCES Restaurant_Tables(table_id)
);




CREATE TABLE IF NOT EXISTS card_payments (
    card_id INT AUTO_INCREMENT PRIMARY KEY,
    account_holder_name VARCHAR(255) NOT NULL,
    card_number VARCHAR(16) NOT NULL,
    expiry_date VARCHAR(7) NOT NULL,
    security_code VARCHAR(3) NOT NULL,
    CONSTRAINT chk_card_number_length CHECK (CHAR_LENGTH(card_number) = 16),
    CONSTRAINT chk_security_code_length CHECK (CHAR_LENGTH(security_code) = 3)
);

CREATE TABLE IF NOT EXISTS Bills (
  bill_id INT PRIMARY KEY AUTO_INCREMENT,
  staff_id INT,
  member_id INT,
  reservation_id INT,
  table_id INT NOT NULL,
  card_id INT,
  payment_method VARCHAR(255),
  bill_time DATETIME NOT NULL,
  payment_time DATETIME,
  INDEX idx_bills_table_payment (table_id, payment_time),
  FOREIGN KEY (staff_id) REFERENCES Staffs(staff_id),
  FOREIGN KEY (member_id) REFERENCES Memberships(member_id),
  FOREIGN KEY (reservation_id) REFERENCES Reservations(reservation_id),
  FOREIGN KEY (table_id) REFERENCES Restaurant_Tables(table_id),
  FOREIGN KEY (card_id) REFERENCES card_payments(card_id)
);




CREATE TABLE IF NOT EXISTS Bill_Items (
  bill_item_id INT PRIMARY KEY AUTO_INCREMENT,
  bill_id INT NOT NULL,
  item_id VARCHAR(6) NOT NULL,
  quantity INT NOT NULL,
  UNIQUE KEY uniq_bill_item (bill_id, item_id),
  CONSTRAINT chk_bill_item_quantity_positive CHECK (quantity > 0),
  FOREIGN KEY (bill_id) REFERENCES Bills(bill_id),
  FOREIGN KEY (item_id) REFERENCES Menu(item_id)
);


CREATE TABLE IF NOT EXISTS Kitchen (
    kitchen_id INT AUTO_INCREMENT PRIMARY KEY,
    table_id INT NOT NULL,
    item_id VARCHAR(6) NOT NULL,
    quantity INT NOT NULL,
    time_submitted DATETIME,
    time_ended DATETIME,
    INDEX idx_kitchen_active_item (table_id, item_id, time_ended),
    CONSTRAINT chk_kitchen_quantity_positive CHECK (quantity > 0),
    FOREIGN KEY (table_id) REFERENCES Restaurant_Tables(table_id),
    FOREIGN KEY (item_id) REFERENCES Menu(item_id)
);

CREATE TABLE IF NOT EXISTS Inventory_Units (
    unit_id INT AUTO_INCREMENT PRIMARY KEY,
    unit_name VARCHAR(50) NOT NULL,
    unit_symbol VARCHAR(20) NOT NULL,
    unit_group VARCHAR(50) NOT NULL,
    UNIQUE KEY uniq_inventory_unit_name (unit_name),
    UNIQUE KEY uniq_inventory_unit_symbol (unit_symbol)
);

CREATE TABLE IF NOT EXISTS Inventory_Categories (
    category_id INT AUTO_INCREMENT PRIMARY KEY,
    category_name VARCHAR(100) NOT NULL,
    storage_type VARCHAR(50) NOT NULL,
    UNIQUE KEY uniq_inventory_category_name (category_name)
);

CREATE TABLE IF NOT EXISTS Inventory_Suppliers (
    supplier_id INT AUTO_INCREMENT PRIMARY KEY,
    supplier_name VARCHAR(255) NOT NULL,
    contact_person VARCHAR(255) NOT NULL,
    phone VARCHAR(20) NOT NULL,
    email VARCHAR(255) NOT NULL,
    address VARCHAR(255) NOT NULL,
    lead_time_days INT NOT NULL DEFAULT 2,
    on_time_score DECIMAL(4,1) NOT NULL DEFAULT 8.0,
    is_active TINYINT(1) NOT NULL DEFAULT 1,
    UNIQUE KEY uniq_supplier_name (supplier_name)
);

CREATE TABLE IF NOT EXISTS Inventory_Ingredients (
    ingredient_id VARCHAR(12) PRIMARY KEY,
    ingredient_name VARCHAR(255) NOT NULL,
    category_id INT NOT NULL,
    purchase_unit_id INT NOT NULL,
    usage_unit_id INT NOT NULL,
    current_stock DECIMAL(12,3) NOT NULL DEFAULT 0,
    reorder_level DECIMAL(12,3) NOT NULL DEFAULT 0,
    par_level DECIMAL(12,3) NOT NULL DEFAULT 0,
    average_unit_cost DECIMAL(10,2) NOT NULL DEFAULT 0,
    storage_area VARCHAR(100) NOT NULL,
    shelf_life_days INT NOT NULL DEFAULT 0,
    is_perishable TINYINT(1) NOT NULL DEFAULT 1,
    is_active TINYINT(1) NOT NULL DEFAULT 1,
    last_stocked_at DATETIME NULL,
    FOREIGN KEY (category_id) REFERENCES Inventory_Categories(category_id),
    FOREIGN KEY (purchase_unit_id) REFERENCES Inventory_Units(unit_id),
    FOREIGN KEY (usage_unit_id) REFERENCES Inventory_Units(unit_id)
);

CREATE TABLE IF NOT EXISTS Inventory_Supplier_Items (
    supplier_item_id INT AUTO_INCREMENT PRIMARY KEY,
    supplier_id INT NOT NULL,
    ingredient_id VARCHAR(12) NOT NULL,
    supplier_price DECIMAL(10,2) NOT NULL,
    min_order_qty DECIMAL(12,3) NOT NULL DEFAULT 1,
    is_preferred TINYINT(1) NOT NULL DEFAULT 0,
    UNIQUE KEY uniq_supplier_ingredient (supplier_id, ingredient_id),
    FOREIGN KEY (supplier_id) REFERENCES Inventory_Suppliers(supplier_id),
    FOREIGN KEY (ingredient_id) REFERENCES Inventory_Ingredients(ingredient_id)
);

CREATE TABLE IF NOT EXISTS Menu_Ingredients (
    recipe_item_id INT AUTO_INCREMENT PRIMARY KEY,
    item_id VARCHAR(6) NOT NULL,
    ingredient_id VARCHAR(12) NOT NULL,
    quantity_required DECIMAL(12,3) NOT NULL,
    waste_percentage DECIMAL(5,2) NOT NULL DEFAULT 0,
    UNIQUE KEY uniq_recipe_item (item_id, ingredient_id),
    FOREIGN KEY (item_id) REFERENCES Menu(item_id),
    FOREIGN KEY (ingredient_id) REFERENCES Inventory_Ingredients(ingredient_id)
);

CREATE TABLE IF NOT EXISTS Purchase_Orders (
    po_id INT AUTO_INCREMENT PRIMARY KEY,
    supplier_id INT NOT NULL,
    order_date DATE NOT NULL,
    expected_date DATE NOT NULL,
    received_at DATETIME NULL,
    status VARCHAR(30) NOT NULL DEFAULT 'ordered',
    subtotal DECIMAL(12,2) NOT NULL DEFAULT 0,
    tax_amount DECIMAL(12,2) NOT NULL DEFAULT 0,
    total_amount DECIMAL(12,2) NOT NULL DEFAULT 0,
    notes VARCHAR(255) DEFAULT NULL,
    created_by INT DEFAULT NULL,
    FOREIGN KEY (supplier_id) REFERENCES Inventory_Suppliers(supplier_id),
    FOREIGN KEY (created_by) REFERENCES Staffs(staff_id)
);

CREATE TABLE IF NOT EXISTS Purchase_Order_Items (
    po_item_id INT AUTO_INCREMENT PRIMARY KEY,
    po_id INT NOT NULL,
    ingredient_id VARCHAR(12) NOT NULL,
    ordered_qty DECIMAL(12,3) NOT NULL,
    received_qty DECIMAL(12,3) NOT NULL DEFAULT 0,
    unit_cost DECIMAL(10,2) NOT NULL,
    line_total DECIMAL(12,2) NOT NULL,
    FOREIGN KEY (po_id) REFERENCES Purchase_Orders(po_id),
    FOREIGN KEY (ingredient_id) REFERENCES Inventory_Ingredients(ingredient_id)
);

CREATE TABLE IF NOT EXISTS Inventory_Stock_Movements (
    movement_id INT AUTO_INCREMENT PRIMARY KEY,
    ingredient_id VARCHAR(12) NOT NULL,
    movement_type VARCHAR(30) NOT NULL,
    quantity DECIMAL(12,3) NOT NULL,
    unit_cost DECIMAL(10,2) NOT NULL DEFAULT 0,
    reference_type VARCHAR(50) DEFAULT NULL,
    reference_id INT DEFAULT NULL,
    staff_id INT DEFAULT NULL,
    notes VARCHAR(255) DEFAULT NULL,
    created_at DATETIME NOT NULL,
    FOREIGN KEY (ingredient_id) REFERENCES Inventory_Ingredients(ingredient_id),
    FOREIGN KEY (staff_id) REFERENCES Staffs(staff_id)
);

CREATE TABLE IF NOT EXISTS Waste_Log (
    waste_id INT AUTO_INCREMENT PRIMARY KEY,
    ingredient_id VARCHAR(12) NOT NULL,
    quantity DECIMAL(12,3) NOT NULL,
    waste_reason VARCHAR(30) NOT NULL,
    notes VARCHAR(255) DEFAULT NULL,
    logged_at DATETIME NOT NULL,
    logged_by INT DEFAULT NULL,
    FOREIGN KEY (ingredient_id) REFERENCES Inventory_Ingredients(ingredient_id),
    FOREIGN KEY (logged_by) REFERENCES Staffs(staff_id)
);

CREATE TABLE IF NOT EXISTS Inventory_Alerts (
    alert_id INT AUTO_INCREMENT PRIMARY KEY,
    ingredient_id VARCHAR(12) NOT NULL,
    alert_type VARCHAR(30) NOT NULL,
    alert_message VARCHAR(255) NOT NULL,
    status VARCHAR(20) NOT NULL DEFAULT 'active',
    created_at DATETIME NOT NULL,
    resolved_at DATETIME DEFAULT NULL,
    UNIQUE KEY uniq_active_alert (ingredient_id, alert_type),
    FOREIGN KEY (ingredient_id) REFERENCES Inventory_Ingredients(ingredient_id)
);
